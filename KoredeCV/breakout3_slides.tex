% Slide 2: Opening Question
\begin{frame}{The Scenario}
    \large
    \textbf{The Scenario:}
    
    \vspace{0.5cm}
    
    Nintendo patented a gameplay mechanic where a character summons another character to fight automatically.
    
    \vspace{0.3cm}
    
    \begin{itemize}
        \item Critics say similar systems existed before
        \item It was approved quickly with no objections
        \item It could be used to threaten competitors
    \end{itemize}
    
    \vspace{0.5cm}
    
    \textbf{\textcolor{UGARed}{Question:}} Should a company be able to patent a common software interaction?
    
    \vspace{0.3cm}
    
    \textbf{Why? Who benefits? Who loses?}
\end{frame}

% Slide 3: Five Statements (CORE SLIDE)
\begin{frame}{Take a Position}
    \Large
    \textbf{Choose one statement to support:}
    
    \vspace{0.8cm}
    
    \begin{enumerate}
        \setlength\itemsep{0.8cm}
        \item Nintendo earned this patent through \textbf{genuine innovation}
        
        \item Games using summoning mechanics should \textbf{license Nintendo's patent—that's how IP protection works}
        
        \item Gameplay mechanics \textbf{shouldn't be patentable at all}
        
        \item Big companies need \textbf{strong patents to prevent cheap knockoffs}
        
        \item The patent office failed—\textbf{prior art existed and this never should have been approved}
    \end{enumerate}
\end{frame}

\begin{frame}{The Counterargument}
    \Large
    
    \vspace{1cm}

    \vspace{0.5cm}
    
    \textbf{Make the case against it.}
    
\end{frame}

% Slide 5: Nintendo Case Deep Dive
\begin{frame}{The Nintendo Case}
    \large
    
    \textbf{Patent 397: Summoning \& Battling Characters}
    
    \vspace{0.5cm}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{What Happened:}
            \begin{itemize}
                \item Filed and approved without objection
                \item Covers summoning a helper character to fight
                \item Legal experts surprised—usually gets pushback
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \textbf{The Criticism:}
            \begin{itemize}
                \item IP lawyer: "Embarrassing failure of the US patent system"
                \item Prior art exists in older games
                \item Could limit competition
            \end{itemize}
        \end{column}
    \end{columns}
    
    \vspace{0.8cm}
    
    \textbf{\textcolor{UGARed}{Discuss:}} Does strong IP protection help or hurt the gaming industry?
\end{frame}
